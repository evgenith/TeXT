% Beamer presentation template. 
%
% No heavy math processing --- just the defaults.
%
% The options set do the following:
%
% serif - Use serif font (default is sans-serif)
% mathserif, professionalfont - Palatino text and Euler math look nice
% with these.
% slidestop - slide content start at top left (default is centered)
\documentclass[english,slidestop]{beamer}

\usepackage[english]{babel}
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}

\usepackage{hyperref}
\hypersetup{
  a4paper,
  bookmarksnumbered,
  bookmarksopen,
  bookmarksopenlevel=0,
  colorlinks,
% for colors, check package xcolor
%   anchorcolor=anchorcolor,
    citecolor=blue,
%   linkcolor=linkcolor,
	pdfauthor={},
	pdftitle={},
	pdfsubject={},
	pdfkeywords={},
%   plainpages=false,
%   urlcolor=urlcolor
}

% Fonts!  
% Concrete is big and readable, while Euler math looks very nice and 
% is easy to distinguish. 
% For more, see http://math.ecnu.edu.cn/~latex/beamer/MathFonts.pdf

% Concrete text and Euler math.
\usepackage{concrete}
\usepackage[euler-digits]{eulervm}

% Serif version please
\usefonttheme{serif}
\usefonttheme{professionalfonts}


% Theme. See http://www.pletscher.org/writings/latex/beamerthemes.php
% for many more.
\usetheme{Boadilla}

% No nav symbols
\beamertemplatenavigationsymbolsempty

% Frame numbers in foot hack:
% If the theme you chose above does not have numbers in the 
% footer, and you want them, these lines may work (try them!).
\expandafter\def\expandafter\insertshorttitle\expandafter{%
  \insertshorttitle\hfill\insertframenumber\,/\,\inserttotalframenumber}


% Standard data: title, author, etc.
\title{}
\author{}
\institute{}

%\date{}


\begin{document}

\maketitle{}

  \frametitle{Outline}
  \tableofcontents

\end{document}
